SELECT
	e.uid AS event_id,
	event_name,
	MIN(s.uid) AS schedule_id,
	date_format(MIN(due_date), '%Y/%m/%d') AS due_date
FROM
	events e
	LEFT JOIN
		schedules s
	ON	e.uid = s.event_id
WHERE
	s.uid = (
		select
			ss.uid
		from
			schedules ss
		where
			ss.event_id = e.uid
		and	due_date >= (
				SELECT
					MAX(sysdate)
				FROM
					dummy_sysdate
			)
		order by
			ss.due_date limit 1
	)
OR	due_date IS NULL
GROUP BY
	e.uid,
	event_name
ORDER BY
	s.event_id IS NULL,
	MIN(due_date),
	e.uid;
