SELECT
	date_format(d.date, '%Y-%m-%d') AS date,
	IFNULL(j.joined_count, 0) AS joined_count,
	IFNULL(l.left_count, 0) AS left_count,
	(
		SELECT
			SUM(ref.j_count) - SUM(ref.l_count)
		FROM
		(
			SELECT
				joined_date AS ref_date,
				1 AS j_count,
				0 AS l_count
			FROM
				users
			UNION ALL
			SELECT
				left_date AS date,
				0 AS j_count,
				1 AS l_count
			FROM
				users
			WHERE
				left_date IS NOT NULL
		) ref
		WHERE
			d.date >= ref.ref_date
	) AS user_count
FROM
(
	SELECT
		joined_date AS date
	FROM
		users
	UNION SELECT
		left_date AS date
	FROM
		users
	WHERE
		left_date IS NOT NULL
) d
LEFT JOIN
(
	SELECT
		joined_date,
		count(*) AS joined_count
	FROM
		users
	GROUP BY
		joined_date
) j
ON
	d.date = j.joined_date
LEFT JOIN
(
	SELECT
		left_date,
		count(*) AS left_count
	FROM
		users
	GROUP BY
		left_date
) l
ON
	d.date = l.left_date
ORDER BY
	d.date;
