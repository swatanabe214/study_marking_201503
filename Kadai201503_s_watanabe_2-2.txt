SELECT
	e.uid,
	e.employee_name,
	sh.section_name,
	date_format(sh.start_date, '%Y/%m/%d') AS start_date
FROM
	employees e
INNER JOIN
	section_histories sh
ON
	sh.employee_id = e.uid
WHERE
	sh.section_name = 'IT'
AND
	sh.start_date =
	(
		SELECT
			MAX(sh2.start_date)
		FROM
			section_histories sh2
		WHERE
			sh.employee_id = sh2.employee_id
	)
ORDER BY
	e.uid;
