select
	e.UID as EMPLOYEE_ID,
	e.EMPLOYEE_NAME,
	h.SECTION_NAME,
	date_format(currentData.maxStartDate, '%Y/%m/%d') as START_DATE
from
	(
		SELECT
			employee_id,
			MAX(start_date) AS maxStartDate
		FROM
			section_histories
		GROUP BY
			employee_id
	) currentData
	LEFT JOIN
		section_histories h
	ON	currentData.employee_id = h.EMPLOYEE_ID
	AND	currentData.maxStartDate = h.START_DATE
	LEFT JOIN
		employees e
	ON	currentData.employee_id = e.uid
where
	SECTION_NAME = 'IT'
order by
	EMPLOYEE_ID;
