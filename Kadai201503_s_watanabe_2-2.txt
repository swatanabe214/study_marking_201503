SELECT
	e.uid AS EMPLOYEE_ID,
	e.EMPLOYEE_NAME,
	sh.SECTION_NAME,
	DATE_FORMAT(sh.start_date, '%Y/%m/%d') AS START_DATE
FROM
	employees e
INNER JOIN
	section_histories sh
ON
	sh.employee_id = e.uid
WHERE
	sh.section_name = 'IT'
AND
	sh.start_date =
	(
		SELECT
			MAX(sh2.start_date)
		FROM
			section_histories sh2
		WHERE
			sh.employee_id = sh2.employee_id
	)
ORDER BY
	e.uid;
